    1. Add a module-level pure helper function (before the class):

    ```typescript
    function applySignature(
      body: string,
      signature: string | undefined,
      isHtml: boolean,
      includeSignature: boolean
    ): string {
      if (!includeSignature || !signature) return body;
      if (isHtml) {
        return `${body}<br><br><p style="white-space: pre-line">-- \n${signature}</p>`;
      }
      return `${body}\n-- \n${signature}`;
    }
    ```

    2. Update `sendEmail()` method signature to add `includeSignature: boolean = true` as the
       last parameter (after `bcc?: string`):

    ```typescript

      isHtml: boolean = false,
      cc?: string,
      bcc?: string,
      includeSignature: boolean = true
    ): Promise<any>
    ```

    3. Inside `sendEmail()`, apply the helper before building `rawMessage` and before calling
       `smtpClient.send()`. Replace the existing `body` usage with `effectiveBody`:

    ```typescript
    const effectiveBody = applySignature(body, this.account.signature, isHtml, includeSignature);

    const rawMessage = [
      `From: ${this.account.user}`,
      `To: ${to}`,
      ...(cc ? [`Cc: ${cc}`] : []),
      ...(bcc ? [`Bcc: ${bcc}`] : []),
      `Subject: ${subject}`,
      'MIME-Version: 1.0',
      `Content-Type: ${isHtml ? 'text/html' : 'text/plain'}; charset=utf-8`,
      '',
      effectiveBody
    ].join('\r\n');

    await this.ensureSmtp();
    const info = await this.smtpClient.send(to, subject, effectiveBody, isHtml, cc, bcc);
    ```
